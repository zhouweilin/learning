####Enviroment Config
```
 npm install webpack -g                       //全局安装webpack    
 npm install react react-dom --save-dev       //开发环境安装 react 和 react-dom
 npm install babel-loader --save-dev          //开发环境安装 babel-loader （会自动安装依赖 babel-core）
 npm install babel-preset-es2015 --save-dev             //开发环境安装 babel-preset-es2015 用于将es6 转换成 es5
 npm install babel-preset-react  --save-dev             //开发环境安装 babel-preset-react 用于将jsx语法转换成普通的js语法
 
```
